```python
# Code to calculate the first 10,000 terms of the series: 1 - 1/3 + 1/5 - 1/7 + ...
total_sum = 0

for i in range(10000):
    term = 1 / (2 * i + 1)  # Calculate the term
    if i % 2 == 0:  # Even index: add the term
        total_sum += term
    else:           # Odd index: subtract the term
        total_sum -= term

# Multiply the total by 4
final_result = total_sum * 4
print(final_result)
```
